How It Works

The isLaunchPage() function checks if the current path includes 'blog/'

// LAUNCH_PAGES array with folder matching:
const LAUNCH_PAGES = [
    'index.html',
    'public/dcf_about.html',
    'public/dcf_contact.html',
    'public/dcf_ai_resources.html',
    'public/dcf_ai_resource_view.html',
    
    'initiatives/',  // ‚Üê Matches ALL files in initiatives folder
    'blog/',         // ‚Üê Matches ALL files in blog folder ‚úÖ
    
    'members/dcf_donate.html'
];

// Function correctly handles folder matching:
if (page.endsWith('/')) {
    const matches = currentPath.includes(page);
    if (matches) {
        console.log(`üìÅ Folder match found: "${page}" in path "${currentPath}"`);
    }
    return matches;
}
